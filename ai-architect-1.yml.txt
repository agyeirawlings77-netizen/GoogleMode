# AI MEMORY SYSTEM
# Persisted to .ai_memory.json in the repo — survives 6h restarts,
# context resets, token limit hits. New runs load it and continue.
# ══════════════════════════════════════════════════════════════════════════
class AIMemory:
    def __init__(self):
        self.data = {
            "total_runs": 0, "total_fixes": 0,
            "phase1_complete": False, "phase2_complete": False,
            "fixes_summary": [], "failed_attempts": [],
            "known_solutions": [], "last_updated": "",
        }
        self._load()

    def _load(self):
        try:
            if AI_MEMORY_FILE.exists():
                self.data = json.loads(AI_MEMORY_FILE.read_text())
                print(f"  📚 Memory: {len(self.data['fixes_summary'])} fixes, "
                      f"{self.data['total_runs']} prior runs.")
        except Exception as e:
            print(f"  ⚠️  Memory load failed: {e} — starting fresh.")

    def _trim(self):
        self.data["fixes_summary"]   = self.data["fixes_summary"][-MAX_MEM_FIXES:]
        self.data["failed_attempts"] = self.data["failed_attempts"][-MAX_MEM_ATTEMPTS:]
        self.data["known_solutions"] = self.data["known_solutions"][-MAX_MEM_SOLS:]

    def save(self):
        self._trim()
        try:
            AI_MEMORY_FILE.write_text(json.dumps(self.data, indent=2))
        except Exception as e:
            print(f"  ⚠️  Memory save: {e}")

    def record_fix(self, phase: str, turn: int, expl: str,
                   files: list, crashes: list):
        self.data["total_fixes"] += 1
        short = expl[:200]
        self.data["fixes_summary"].append({
            "phase": phase, "turn": turn, "explanation": short,
            "files": files[:10], "crashes": crashes[:5],
            "at": time.strftime("%Y-%m-%d %H:%M"),
        })
        if short not in self.data["known_solutions"]:
            self.data["known_solutions"].append(short)
        self.data["last_updated"] = time.strftime("%Y-%m-%d %H:%M UTC")
        self.save()

    def record_failed(self, phase: str, turn: int, expl: str):
        entry = f"[{phase} t{turn}] {expl[:150]}"
        if entry not in self.data["failed_attempts"]:
            self.data["failed_attempts"].append(entry)
        self.save()

    def mark_done(self, phase: str):
        if phase == "1": self.data["phase1_complete"] = True
        if phase == "2": self.data["phase2_complete"] = True
        self.save()

    def inc_run(self):
        self.data["total_runs"] += 1
        # FIX 3: infinite restart guard — stops burning GitHub minutes
        # if something is fundamentally broken (e.g. build always fails)
        MAX_TOTAL_RUNS = 50
        if self.data["total_runs"] > MAX_TOTAL_RUNS:
            print(f"\n🛑 SAFETY STOP: total_runs={self.data['total_runs']} "
                  f"> {MAX_TOTAL_RUNS} — too many restarts.")
            print("  This means something is fundamentally broken.")
            print("  Check your code, Firebase config, or Gradle setup manually.")
            print("  ai-sandbox branch has all progress committed for inspection.")
            sys.exit(1)
        self.save()

    def build_context(self) -> str:
        """
        Compact memory block injected into EVERY AI prompt.
        Trimmed to last 15 fixes, 10 failed, 10 working.
        Explanations capped at 200 chars to prevent token bloat.
        """
        lines = [
            "━━━━━━━━ AI MEMORY (READ FIRST) ━━━━━━━━",
            f"Runs:{self.data['total_runs']} | "
            f"Fixes:{self.data['total_fixes']} | "
            f"P1:{'✅' if self.data['phase1_complete'] else '❌'} | "
            f"P2:{'✅' if self.data['phase2_complete'] else '❌'}",
        ]
        if self.data["fixes_summary"]:
            lines.append("\nALREADY FIXED (do NOT redo):")
            for f in reversed(self.data["fixes_summary"][-15:]):
                lines.append(f"  [{f['phase']} t{f['turn']}] {f['explanation']}")
                if f.get("crashes"):
                    lines.append(f"    ↳ {', '.join(f['crashes'][:3])}")
        if self.data["failed_attempts"]:
            lines.append("\n⛔ FAILED APPROACHES (do NOT repeat):")
            for fa in self.data["failed_attempts"][-10:]:
                lines.append(f"  • {fa}")
        if self.data["known_solutions"]:
            lines.append("\n✅ APPROACHES THAT WORKED:")
            for ks in self.data["known_solutions"][-10:]:
                lines.append(f"  • {ks}")
        lines.append("━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━\n")
        return "\n".join(lines)

_restart_lock = threading.Lock()   # prevent double-restart from watchdog + main

def restart_ai(reason: str) -> None:
    """
    THE CORE BRAIN — called from:
    - Watchdog thread (time limit, runs independently of main loop)
    - Main loop (context overflow, memory cap, stall, token limit)
    Thread-safe via _restart_lock so only one restart ever fires.
    """
    with _restart_lock:
        print(f"\n🔁 RESTARTING AI — Reason: {reason}")
        print(f"   Elapsed: {(time.monotonic()-RUN_START)/60:.1f} min")

        memory.save()
        print("  📚 Memory saved.")

            _git(f"git rm --cached GooglePro/{f} 2>/dev/null || true")
        _git("git add -A")
        cr = subprocess.run(
            ["git", "commit", "-F", "-"],
            input=f"AI restart checkpoint ({reason})",
            capture_output=True, text=True, cwd=str(GIT_ROOT)
        )
        if cr.returncode == 0:
            print("  ✅ Checkpoint committed.")
        else:
            print("  ℹ️  Nothing new to commit.")

        # Step 3: TRIGGER A — push to ai-sandbox (primary re-trigger)
        # Workflow listens to ai-sandbox pushes, so this directly re-fires it
        push = subprocess.run(
            ["git", "push", "origin", "ai-sandbox", "--force"],
            capture_output=True, text=True, cwd=str(GIT_ROOT)
        )
        trigger_a = push.returncode == 0
        print(f"  {'📤 Push trigger fired' if trigger_a else '⚠️  Push failed'}: ai-sandbox")

        # Step 4: TRIGGER B — workflow_dispatch API (fallback)
        # Fires even if push had no new commits to push
        trigger_b = False
        if GH_TOKEN and REPO:
            url = (f"https://api.github.com/repos/{REPO}/actions"
                   f"/workflows/{WORKFLOW_FILE}/dispatches")
            try:
                req = urllib.request.Request(
                    url, data=json.dumps({"ref": "ai-sandbox"}).encode(),
                    headers={
                        "Authorization": f"Bearer {GH_TOKEN}",
                        "Accept": "application/vnd.github+json",
                        "Content-Type": "application/json",
                        "X-GitHub-Api-Version": "2022-11-28",
                    }, method="POST")
                with urllib.request.urlopen(req, timeout=30) as r:
                    print(f"  🚀 workflow_dispatch fired (HTTP {r.status}) — fallback ready.")
                    trigger_b = True
            except Exception as e:
                print(f"  ⚠️  workflow_dispatch failed: {e}")

        # Step 5: report outcome
        if trigger_a or trigger_b:
            triggered_by = []
            if trigger_a: triggered_by.append("push→ai-sandbox")
            if trigger_b: triggered_by.append("workflow_dispatch")
            print(f"  ✅ New run guaranteed via: {' + '.join(triggered_by)}")
        else:
            print("  ❌ Both triggers failed — run manually from Actions tab.")
            print("  ℹ️  Progress is saved. Go to Actions → Run workflow.")

        print("  🛑 Forcing process exit — no duplicate loops possible.")
        # os._exit(0) kills the ENTIRE process immediately including all threads.
        # sys.exit(0) only raises SystemExit in the calling thread — if called
        # from the watchdog thread, the main thread keeps running. os._exit(0)
        # guarantees everything stops. GitHub Action job ends cleanly.
        os._exit(0)

# ══════════════════════════════════════════════════════════════════════════
# WATCHDOG THREAD — fires restart_ai() at SOFT_LIMIT regardless of what
# the main thread is doing (blocking build, slow AI call, long analyze).
# This is the fix for "time check only runs between turns".
# ══════════════════════════════════════════════════════════════════════════
def _watchdog():
    """
    Background daemon thread. Completely independent of the main thread.
    Fires restart_ai() at SOFT_LIMIT_SECS regardless of what main is doing:
    - long flutter build (can take 30+ min)
    - slow AI API response
    - stuck flutter analyze
    Checks every 30 seconds for accuracy. os._exit(0) in restart_ai()
    guarantees the entire process stops — no race conditions.
    """
    while True:
        elapsed   = time.monotonic() - RUN_START
        remaining = SOFT_LIMIT_SECS - elapsed
        if remaining <= 0:
            print("\n⏰ WATCHDOG: Time limit reached — firing restart_ai()")
            restart_ai("watchdog_6h_time_limit")
            return  # os._exit(0) inside restart_ai means this never runs
        # Log a warning when under 30 minutes remaining
        if remaining < 1800:
            print(f"  ⏰ Watchdog: {remaining/60:.1f} min remaining before restart")
        time.sleep(min(30, remaining))

# Start watchdog as daemon thread immediately
_wdog = threading.Thread(target=_watchdog, daemon=True, name="watchdog")
_wdog.start()

    Inline checks at each loop turn. The watchdog handles the time limit
    independently, but these catch memory cap and stall conditions.
    """
    global _last_repo_hash

    # Memory cap check
    if len(memory.data["fixes_summary"]) >= MAX_MEM_FIXES:
        restart_ai(f"memory_cap ({MAX_MEM_FIXES}) at {label}")

    # Stall detection — if repo hash hasn't changed, nothing was committed
    current_hash = get_repo_hash()
    if _last_repo_hash is not None and current_hash == _last_repo_hash:
        print(f"  ⚠️  No repo changes since last turn at {label}.")
    _last_repo_hash = current_hash

                          cwd=str(cwd or REPO_ROOT))

    return run(cmd, cwd=GIT_ROOT)

def git_commit(msg: str) -> None:
    for f in BLOCKED:
        _git(f"git rm --cached GooglePro/{f} 2>/dev/null || true")
    _git("git add -A")
    if _git("git diff --cached --quiet").returncode != 0:
        r = subprocess.run(["git", "commit", "-F", "-"], input=msg,
                           capture_output=True, text=True, cwd=str(GIT_ROOT))
        print(f"  {'✅' if r.returncode==0 else '❌'} Commit: {msg[:72]}")
    else:
        print("  Nothing to commit.")

# RESPONSE VALIDATION
# Called before parse_json — catches all AI failure modes.
# ══════════════════════════════════════════════════════════════════════════
class AIEmptyResponse(Exception):   pass
class AITruncatedOutput(Exception): pass
class AITokenLimitHit(Exception):   pass

def validate_response(text: str, phase: str, turn: int) -> None:
    # ① Empty / near-empty
    if not text or len(text.strip()) < 50:
        raise AIEmptyResponse(
            f"AI_EMPTY phase={phase} turn={turn} len={len(text) if text else 0}")

    tl = text.lower()

    # ② Token limit string in response body
    if "max_tokens" in tl and len(text) < 200:
        raise AITokenLimitHit(f"TOKEN_LIMIT_IN_BODY phase={phase} turn={turn}")

    # ③ Context error keywords in response body
    if any(kw in tl for kw in CONTEXT_KEYWORDS):
        raise AITokenLimitHit(f"CONTEXT_ERROR_IN_TEXT phase={phase} turn={turn}")

    # ④ JSON opened but never closed (truncated mid-generation)
    opens  = text.count("{")
    closes = text.count("}")
    if opens > 0 and closes == 0:
        raise AITruncatedOutput(
            f"TRUNCATED phase={phase} turn={turn} {opens}↑ 0↓")
    if opens > 0 and opens > closes + 3:
        raise AITruncatedOutput(
            f"TRUNCATED phase={phase} turn={turn} {opens}↑ {closes}↓")

    print(f"  ✅ Response: {len(text):,} chars | JSON {opens}↑/{closes}↓")

# API CALLER — with stop_reason check, response validation, auto-restart
# ══════════════════════════════════════════════════════════════════════════
def call_ai(messages: list, system: str,
            phase: str = "?", turn: int = 0,
            retries: int = 3) -> str:
    payload = json.dumps({
        "model": MODEL, "max_tokens": 8192,
        "system": system, "messages": messages,
    }).encode()
    hdrs = {
        "x-api-key": API_KEY,
        "anthropic-version": "2023-06-01",
        "content-type": "application/json",
    }
    last_exc: Optional[Exception] = None

    for attempt in range(retries):
        try:
            req = urllib.request.Request(API_URL, data=payload,
                                         headers=hdrs, method="POST")
            with urllib.request.urlopen(req, timeout=120) as r:
                raw       = r.read()
                resp_data = json.loads(raw)

                # Check stop_reason from API metadata
                stop_reason = resp_data.get("stop_reason", "")
                usage       = resp_data.get("usage", {})
                out_tokens  = usage.get("output_tokens", 0)

                if stop_reason == "max_tokens":
                    print(f"  ⚠️  stop_reason=max_tokens ({out_tokens} tokens) — truncated.")
                    memory.record_failed(phase, turn, "stop_reason_max_tokens")
                    restart_ai("stop_reason_max_tokens")

                text = resp_data["content"][0]["text"]

                # Validate response content before returning
                validate_response(text, phase, turn)
                return text

        except (AIEmptyResponse, AITruncatedOutput, AITokenLimitHit) as exc:
            # Validation failures → restart immediately, no retry
            print(f"  ⚠️  Response invalid: {exc}")
            memory.record_failed(phase, turn, str(exc))
            restart_ai(str(exc))

        except urllib.error.HTTPError as exc:
            body = exc.read().decode(errors="ignore")
            err  = body.lower()
            print(f"  HTTP {exc.code}: {body[:300]}")
            last_exc = exc

            # Context overflow via HTTP error
            if exc.code in (400, 413, 422):
                if any(kw in err for kw in CONTEXT_KEYWORDS):
                    print(f"  ⚠️  Context overflow (HTTP {exc.code}).")
                    memory.record_failed(phase, turn, "context_overflow")
                    restart_ai(f"context_overflow_HTTP_{exc.code}")

            wait = 30 * (attempt + 1) if exc.code in (429, 529) else 10
            print(f"  Retry {attempt+1}/{retries} in {wait}s…")
            time.sleep(wait)

        except Exception as exc:
            err = str(exc).lower()
            print(f"  Error: {exc}")
            last_exc = exc
            if any(kw in err for kw in CONTEXT_KEYWORDS):
                memory.record_failed(phase, turn, "context_overflow_exception")
                restart_ai("context_overflow_exception")
            time.sleep(10)

    raise last_exc or RuntimeError("call_ai exhausted all retries")

# ══════════════════════════════════════════════════════════════════════════
# JSON PARSER
# ══════════════════════════════════════════════════════════════════════════
def parse_json(text: str) -> dict:
    text = re.sub(r"^```(?:json)?\s*", "", text.strip())
    text = re.sub(r"\s*```$", "", text)
    s = text.find("{")
    if s == -1: raise ValueError("No JSON object found")
    depth, end = 0, -1
    for i, ch in enumerate(text[s:], s):
        if ch == "{":   depth += 1
        elif ch == "}":
            depth -= 1
            if depth == 0: end = i; break
    if end == -1: raise ValueError("Unbalanced JSON braces")
    return json.loads(text[s:end+1])

def read_file(path: str) -> Optional[str]:
    try:
        with open(REPO_ROOT / path, errors="ignore") as f:
            return f.read()
    except OSError:
        return None

def write_file(rel_path: str, content: str) -> bool:
    """Write with full safety guards."""
    if rel_path in BLOCKED:
        print(f"  🔒 Blocked secret: {rel_path}"); return False
    if len(content) > MAX_CONTENT:
        print(f"  ⚠️  Oversized ({len(content):,}c): {rel_path}"); return False
    if len(content.strip()) < 20:
        print(f"  ⚠️  Empty: {rel_path}"); return False
    try:
        resolved = (REPO_ROOT / rel_path).resolve()
        resolved.relative_to(REPO_ROOT)
    except ValueError:
        print(f"  ⛔ Unsafe path: {rel_path!r}"); return False
    resolved.parent.mkdir(parents=True, exist_ok=True)
    resolved.write_text(content)
    print(f"    ✏️  {rel_path} ({len(content):,}c)")
    return True

def collect_dart_ctx(logs: str) -> dict:
    ctx: dict = {}
    for pat in [r"(?:lib|test|integration_test)/[\w./-]+\.dart",
                r"[\w.-]+\.yaml"]:
        for p in set(re.findall(pat, logs)):
            if any(b in p for b in BLOCKED): continue
            if p not in ctx:
                c = read_file(p)
                if c: ctx[p] = c[:MAX_FILE_CHARS]
    for a in ("pubspec.yaml", "lib/main.dart", "analysis_options.yaml"):
        if a not in ctx:
            c = read_file(a)
            if c: ctx[a] = c[:MAX_FILE_CHARS]
    return ctx

def collect_build_ctx(logs: str) -> dict:
    ctx: dict = {}
    for p in ["pubspec.yaml", "android/app/build.gradle", "android/build.gradle",
              "android/gradle.properties", "android/settings.gradle",
              "android/app/src/main/AndroidManifest.xml",
              "android/app/proguard-rules.pro"]:
        c = read_file(p)
        if c: ctx[p] = c[:MAX_FILE_CHARS]
    for p in set(re.findall(r"(?:lib|test)/[\w./-]+\.dart", logs)):
        if p not in ctx:
            c = read_file(p)
            if c: ctx[p] = c[:MAX_FILE_CHARS]
    return ctx

def validate_dart(written: list) -> bool:
    darts = [p for p in written if p.endswith(".dart")]
    if not darts: return True
    print("  🔍 Validating Dart files…")
    r = run(f"flutter analyze {' '.join(darts)}")
    if r.returncode == 0: print("  ✅ Dart valid."); return True
    print(f"  ⚠️  Issues:\n{(r.stdout+r.stderr)[-1500:]}")
    return False

def ensure_signing() -> None:
    """Inject signing config into build.gradle if not already present."""
    gp = REPO_ROOT / "android/app/build.gradle"
    if not gp.exists(): return
    c = gp.read_text(errors="ignore")
    if "signingConfig signingConfigs.release" in c:
        print("  ✅ Signing already present."); return
    block = """
    signingConfigs {
        release {
            def props = new Properties()
            def f = rootProject.file("key.properties")
            if (f.exists()) props.load(new FileInputStream(f))
            keyAlias      props['keyAlias']      ?: System.getenv('KEY_ALIAS')
            keyPassword   props['keyPassword']   ?: System.getenv('KEY_PASSWORD')
            storeFile     props['storeFile'] ? file(props['storeFile']) : null
            storePassword props['storePassword'] ?: System.getenv('STORE_PASSWORD')
        }
    }
"""
    if "buildTypes {" in c and "signingConfigs {" not in c:
        c = c.replace("buildTypes {", block + "\n    buildTypes {", 1)
    c = re.sub(r"(release\s*\{)",
               r"\1\n            signingConfig signingConfigs.release", c, count=1)
    c = re.sub(r"compileSdk\s+\d+",             f"compileSdk {COMPILE_SDK}",  c)
    c = re.sub(r"minSdk(?:Version)?\s+\d+",     f"minSdk {MIN_SDK}",          c)
    c = re.sub(r"targetSdk(?:Version)?\s+\d+",  f"targetSdk {TARGET_SDK}",    c)
    c = re.sub(r'buildToolsVersion\s+"[\d.]+"',
               f'buildToolsVersion "{BUILD_TOOLS}"', c)
    if "multiDexEnabled" not in c:
        c = re.sub(r"(defaultConfig\s*\{)",
                   r"\1\n        multiDexEnabled true", c, count=1)
    gp.write_text(c)
    print(f"  ✅ Signing injected: minSdk={MIN_SDK} targetSdk={TARGET_SDK}")

            build_msgs.pop()

            consecutive_no_change += 1

            memory.record_failed("p2", i, f"no changes: {expl}")
            if consecutive_no_change >= 2:
                restart_ai("no_progress_phase2")
            build_msgs.append({"role": "user", "content":
                f"Turn {i}: you made NO file changes. "
                "Read the build log carefully and fix actual Gradle/manifest errors."})
            build_msgs.pop(); build_msgs.pop()
            continue
        consecutive_no_change = 0

        written = [p for p, c in fixed_files.items() if write_file(p, c)]
        print(f"  Files written: {len(written)}")

        # Re-run pub get if pubspec changed
        if any("pubspec.yaml" in p for p in written):
            print("  pubspec.yaml changed → flutter pub get…")
            pub = run("flutter pub get")
            if pub.returncode != 0:
                print(f"  ⚠️  pub get failed:\n{(pub.stdout+pub.stderr)[-400:]}")
                memory.record_failed("p2", i, expl)
                build_msgs.append({"role": "user", "content":
                    f"Turn {i}: pubspec change broke pub get:\n"
                    f"{pub.stdout+pub.stderr}\nFix the dependency conflict."})
                _git("git reset --hard HEAD")
                build_msgs.pop(); build_msgs.pop()
                continue

        memory.record_fix("p2", i, expl, written, [])
        git_commit(f"p2-t{i}: {expl}")

    restart_ai("p2_max_iterations")
